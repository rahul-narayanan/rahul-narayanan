MERGE INTO user_payload AS target
      USING (
        VALUES
          ${payloads.map(p => `(${p.id}, '${p.user_email}', ...)`) // Add all columns
      ) AS source (id, user_email, ...)
      ON target.id = source.id
      WHEN MATCHED THEN
        UPDATE SET target.column_to_update = source.column_to_update
      WHEN NOT MATCHED THEN
        INSERT (id, user_email, ...)
        VALUES (source.id, source.user_email, ...);
