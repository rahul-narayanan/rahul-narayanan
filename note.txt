SELECT
    pr.product_role_name AS product_role,
    JSON_QUERY(
        '[' + STRING_AGG('"' + up.data_asset_view_name + '"', ',') + ']'
    ) AS data_asset_names
FROM
    UserPayload up
JOIN
    product_role pr ON pr.product_role_name = up.product_role_name
WHERE
    up.user_email = 'your_user_email_here'
GROUP BY
    pr.product_role_name;
